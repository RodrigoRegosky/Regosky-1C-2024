55 void ShowDistanceTask(void *pvParameter)

57 uint16_t distancia;
58
59 while (true)
60 {
61 if (medir == true)
62 {
63 distancia = HcSr04ReadDistanceInCentimeters();
64 if (distancia < 10)
65 {
66 LedOff(LED_1);
67 LedOff(LED_2);
68 LedOff(LED_3);
69 }
70 else if ((distancia >= 10) & (distancia < 20))
71 {
72 LedOn(LED_1);
73 LedOff(LED_2);
74 LedOff(LED_3);
75 }
76 else if ((distancia >= 20) & (distancia < 30))
77 {
78 LedOn(LED_1);
