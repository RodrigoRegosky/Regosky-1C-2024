196 {
197 Imin = voltajeSensor;
198 }
199
200 if (voltajeSensor > Imax)

203 }
204
205 if (voltajeSensor < Imin)
206 {
207 Imin = voltajeSensor;
208 }
209
210 contador_corriente++;

241void read_data(uint8_t *data, uint8_t length)
242{
243 uint8_t i = 1;
244 static uint8_t time = 0;
245 static uint32_t pot = 0;
246
247 if (data[0] == 'O')
248 {
249 ActivaRelee();
250 ON_OFF = true;
251 }
252
253 if (data[0] == 'o')
254 {
255 DesactivaRelee();
256 ON_OFF = false;
257 consumo = 0;
258 temporizador = 0;
259 }
260
261 if (data[0] == 'T')
262 {
263 /* El slidebar de tiempo envía los datos con el formato "T" + value + "t" */
264 time = 0;
265 while (data[i] != 't')
266 {
267 /* Convertir el valor ASCII a un valor entero */
268 time = time * 10;
269 time = time + (data[i] - '0');
270 i++;
271 }
272 tiempoApagado = time;
273 }
274
275 else if (data[0] == 'P')
276 {
277 /* El slidebar de potencia envia en formato "P" + value + "p" */
278 pot = 0;
279 while (data[i] != 'p')
280 {
281 /* Convertir el valor ASCII a un valor entero */
282 pot = pot * 10;
283 pot = pot + (data[i] - '0');
284 i++;
285 }
286 consumo_maximo = pot;
287 }
