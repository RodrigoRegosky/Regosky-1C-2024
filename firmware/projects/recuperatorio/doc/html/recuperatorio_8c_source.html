90void medirDistancia()
91{
92 while (true)
93 {
94
95 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
96
97 distancia = HcSr04ReadDistanceInCentimeters();
98 if (distancia < 8)
99 {
100 LedOn(LED_1);
101 LedOff(LED_2);
102 LedOff(LED_3);
103
104 distanciaMedicion = false;
105 }
106 else if (distancia > 12)
107 {
108 LedOff(LED_1);
109 LedOff(LED_2);
110 LedOn(LED_3);
111
112 distanciaMedicion = false;
113
114 if (distancia > 140)
115 {
116 nuevoCicloDeMedidas = true; // Si detecta que se alejo mas de 140 cm se considera un nuevo ciclo
117 GPIOOff(GPIO_9); // se apaga la alarma en caso de que estuviese prendida
118 }
119 }
120 else if ((distancia <= 12) && (distancia >= 8))
121 {
122 LedOff(LED_1);
123 LedOn(LED_2);
124 LedOff(LED_3);
125
126 distanciaMedicion = true;
127 }
128 }
129}

138void medirTemperatura()
139{
140 while (true)
141 {
142
143 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
144
145 if (distanciaMedicion == true) // Comprueba si la distancia esta dentro del rango
146 {
147
148 if (nuevoCicloDeMedidas == true) // comprueba que sea la primer medicion que se realiza del ciclo
149 {
150 AnalogInputReadSingle(CH1, &voltaje); // Lee el voltaje de la termopila
151 sumatoriaTemperatura = ((voltaje + 2200) / 110); // Convierte voltaje a temperatura
152 contadorTemperatura++;
153
154 if (contadorTemperatura == 10) // Cada 10 mediciones entra en el ciclo
155 {
156 temperatura = (sumatoriaTemperatura / 10); // calcula el promedio de las 10 mediciones
157
158 tempInt = (int)temperatura;
159
160 // Saca por el puerto serie de la forma "temperatura] Cº persona a [distancia] cm"
161 UartSendString(UART_PC, (char *)UartItoa(tempInt, 10));
162 UartSendString(UART_PC, " Cº persona a ");
163 UartSendString(UART_PC, (char *)UartItoa(distancia, 10));
164 UartSendString(UART_PC, " cm ");
165 UartSendString(UART_PC, "\n\r");
166
167 // Comprueba si la temperatura amerita prender la alarma
168 if (temperatura <= 37.5)
169 {
170 GPIOOff(GPIO_9);
171 }
172 else if (temperatura > 37.5)
173 {
174 GPIOOn(GPIO_9);
175 }
176
177 temperatura = 0;
178 contadorTemperatura = 0;
179 sumatoriaTemperatura = 0;
180 }
181
182 nuevoCicloDeMedidas = false; // Una vez realizada la medicion y mostrar por el puerto serie, se setea a falso para no realizar mas mediciones hasta el siguiente ciclo
183 }
184 }
185 }
186}
