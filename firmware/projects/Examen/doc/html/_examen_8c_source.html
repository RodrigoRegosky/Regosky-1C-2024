338void Sistema()
339{
340 while (true)
341 {
342
343 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
344
345 UartReadByte(UART_PC, &teclaSistema);
346
347 if (teclaSistema == 1)
348 {
349 if (estadoSistema = false)
350 {
351 TimerReset(TIMER_A);
352 TimerReset(TIMER_B);
353 estadoSistema = true;
354 }
355 }
356 else if (teclaSistema == 2)
357 {
358 if (estadoSistema = true)
359 {
360 if (estadoBombaPhA = true)
361 {
362 controlPhA();
363 }
364 if (estadobombaPhB = true)
365 {
366 controlPhB();
367 }
368 if (estadoBombaRiego = true)
369 {
370 controlBombaRiego();
371 }
372
373 TimerStop(TIMER_A);
374 TimerStop(TIMER_B);
375 estadoSistema = false;
376 }
377 }
378 }
