331void Sistema()
332{
333 while (true)
334 {
335
336 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
337
338 UartReadByte(UART_PC, &teclaSistema);
339
340 if (teclaSistema == 1)
341 {
342 if (estadoSistema = false)
343 {
344 TimerReset(TIMER_A);
345 TimerReset(TIMER_B);
346 estadoSistema = true;
347 }
348 }
349 else if (teclaSistema == 2)
350 {
351 if (estadoSistema = true)
352 {
353 if (estadoBombaPhA = true)
354 {
355 controlPhA();
356 }
357 if (estadobombaPhB = true)
358 {
359 controlPhB();
360 }
361 if (estadoBombaRiego = true)
362 {
363 controlBombaRiego();
364 }
365
366 TimerStop(TIMER_A);
367 TimerStop(TIMER_B);
368 estadoSistema = false;
369 }
370 }
371 }
